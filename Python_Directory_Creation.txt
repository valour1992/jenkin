import os
filepath ="D:test"
access_rights =777
dirname=input("enter the name folk:")
folder = os.path.join(filepath,dirname)
if not os.path.exists(folder):
    os.makedirs(folder, access_rights)
    format ="txt"
    filename=dirname
    filename = filename +format
    filepath=os.path.join(folder,filename)
    if not os.path.exists(filepath):
        k=open(filepath,"w")
    txt =input("enter text")
    k.write(txt)
    k.close()
else:
        print("Directory present")
